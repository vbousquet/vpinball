# plugin CMake file to be included in main project CMakelists

# --- Pin2K plugin ---

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_C_STANDARD 99)

set(PIN2K_PLUGIN_SOURCES
   plugins/pin2k/common.cpp
   plugins/pin2k/Pin2KPlugin.cpp

   plugins/pin2k/common.h
)

if(BUILD_SHARED)
   add_library(Pin2KPlugin MODULE ${PIN2K_PLUGIN_SOURCES})

   target_include_directories(Pin2KPlugin PUBLIC
      ${CMAKE_SOURCE_DIR}/third-party/include
      plugins
      plugins/pin2k
   )

   if(PluginPlatform STREQUAL "windows" OR PluginPlatform STREQUAL "windows-mingw")
      if(MSVC)
         add_compile_options(
            $<$<CONFIG:RELEASE>:/Ob2>
            $<$<CONFIG:RELEASE>:/O2>
            $<$<CONFIG:RELEASE>:/Oi>
            $<$<CONFIG:RELEASE>:/arch:SSE2>
            $<$<CONFIG:RELEASE>:/fp:fast>
            $<$<CONFIG:RELEASE>:/fp:except->
            $<$<CONFIG:RELEASE>:/Ot>
            $<$<CONFIG:RELEASE>:/GF>
            $<$<CONFIG:RELEASE>:/GS->
            $<$<CONFIG:RELEASE>:/Gy>
            $<$<CONFIG:RELEASE>:/GR>
            $<$<CONFIG:RELEASE>:/Oy>
            $<$<CONFIG:RELEASE>:/GT>
            $<$<CONFIG:RELEASE>:/GL>
         )
         set_target_properties(Pin2KPlugin PROPERTIES
            MSVC_RUNTIME_LIBRARY "MultiThreaded$<$<CONFIG:DEBUG>:Debug>"
         )
         target_compile_options(Pin2KPlugin PUBLIC
            $<$<CONFIG:RELEASE>:$<$<COMPILE_LANGUAGE:CXX>:${OPT_COMMON}>>
            $<$<CONFIG:RELEASE>:$<$<COMPILE_LANGUAGE:C>:${OPT_COMMON}>>
         )
         target_link_options(Pin2KPlugin PUBLIC
            $<$<CONFIG:RELEASE>:/INCREMENTAL:NO>
            $<$<CONFIG:RELEASE>:/OPT:REF>
            $<$<CONFIG:RELEASE>:/OPT:ICF>
            $<$<CONFIG:RELEASE>:/LTCG>
         )
      endif()
      # Wait, is there a pin2k dll? Encore generates an executable or a dll?
      # The submodule Encore-Pinball2000 is a library or what?
      # I should just set the output name and properties for now.
      if (PluginArch STREQUAL "x64")
         set_target_properties(Pin2KPlugin PROPERTIES
            PREFIX ""
            OUTPUT_NAME "plugin-pin2k64"
         )
      else()
         set_target_properties(Pin2KPlugin PROPERTIES
            PREFIX ""
            OUTPUT_NAME "plugin-pin2k"
         )
      endif()
   else()
      set_target_properties(Pin2KPlugin PROPERTIES
         PREFIX ""
         OUTPUT_NAME "plugin-pin2k"
      )
      if(PluginPlatform STREQUAL "macos")
         set_target_properties(Pin2KPlugin PROPERTIES
            SUFFIX ".dylib"
         )
      endif()
   endif()

   add_custom_command(TARGET Pin2KPlugin POST_BUILD
      COMMAND "${CMAKE_COMMAND}" -E copy "${CMAKE_SOURCE_DIR}/plugins/pin2k/plugin.cfg" "${PLUGINS_DIR}/pin2k/plugin.cfg"
      COMMAND "${CMAKE_COMMAND}" -E copy "$<TARGET_FILE:Pin2KPlugin>" "${PLUGINS_DIR}/pin2k"
      COMMAND "${CMAKE_COMMAND}" -E rm "$<TARGET_FILE:Pin2KPlugin>"
   )

endif()

if(BUILD_STATIC)
   add_compile_options(-fPIC)

   add_library(Pin2KPlugin STATIC ${PIN2K_PLUGIN_SOURCES})

   target_include_directories(Pin2KPlugin PUBLIC
      ${CMAKE_SOURCE_DIR}/third-party/include
      plugins
      plugins/pin2k
   )

   set_target_properties(Pin2KPlugin PROPERTIES
      PREFIX ""
      OUTPUT_NAME "plugin-pin2k"
   )
endif()
